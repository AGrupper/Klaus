    ```python
    def _post(url: str, token: str, payload: dict) -> tuple[int, dict]:
        try:
            import requests  # type: ignore
        except ImportError:
            print("ERROR: requests not installed; pip install requests", file=sys.stderr)
            return 0, {}
        resp = requests.post(
            url,
            headers={"Authorization": f"Bearer {token}", "Content-Type": "application/json"},
            json=payload,
            timeout=30,
        )
        try:
            body = resp.json()
        except Exception:
            body = {"raw": resp.text[:500]}
        return resp.status_code, body


    def _verify_firestore(ts_marker: int, count: int) -> list[str]:
        from memory.firestore_db import MealStore
        today = datetime.now(ZoneInfo("Asia/Jerusalem")).date().isoformat()
        meals = MealStore().get_day(today)
        prefix = f"healthkit:test-{ts_marker}-"
        found = [m for m in meals if m.get("source_id", "").startswith(prefix)]
        if len(found) != count:
            print(f"MISMATCH: posted {count} samples, found {len(found)} Firestore docs with prefix {prefix}")
        else:
            print(f"OK: found {count} Firestore docs with prefix {prefix}")
        return [m["source_id"] for m in found]
    ```

    ```python
    def main() -> int:
        args = _parse_args()
        token = os.environ.get(args.token_env, "")
        if not token:

        payload, ts_marker = _build_synthetic_payload(args.count)
        status, body = _post(args.url, token, payload)
        print(f"POST {args.url} → {status} {body}")
        if status != 200:
            return 1
        source_ids = _verify_firestore(ts_marker, args.count)
        if len(source_ids) != args.count:
            return 1
        if args.no_cleanup:
            print("--no-cleanup set; leaving test docs in place")
            return 0
        ans = input(f"Delete {len(source_ids)} test docs? [y/N]: ").strip().lower()
        if ans == "y":
            from memory.firestore_db import MealStore
            store = MealStore()
            today = datetime.now(ZoneInfo("Asia/Jerusalem")).date().isoformat()
            for sid in source_ids:
                try:
                    store.delete(today, sid)  # if no .delete, use the raw Firestore client
                except Exception as exc:
                    print(f"WARN: could not delete {sid}: {exc}")
            print(f"Deleted {len(source_ids)} test docs.")
    ```

    ## 6. Security Considerations

    - **Authorization header ONLY — NEVER in URL query.** Placing the token in
      the URL query string would leak it via Cloud Run access logs. Keep it in
      the `Authorization: Bearer …` header.
    - **TLS-only.** iOS `Get Contents of URL` does not support cert pinning;
      Cloud Run enforces HTTPS by default; system root CAs are trusted.
    - **Token entropy ≥ 32 bytes** (see DEPLOYMENT.md §23 mint command).
    - **Kill-switch:** if the token leaks, `gcloud secrets versions disable
      klaus-healthkit-webhook-token --version=<n>` instantly rejects all inbound
      pushes. No redeploy required.
    - **No replay protection** (acceptable risk): MealStore.upsert is idempotent
      on source_id, so replayed payloads produce no state delta.
